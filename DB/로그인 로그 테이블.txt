CREATE TABLE LOGIN_LOG (
    LOG_ID       NUMBER PRIMARY KEY,
    EMAIL        VARCHAR2(200),
    STATUS       VARCHAR2(20),       -- SUCCESS / FAIL / LOCKED
    IP_ADDRESS   VARCHAR2(50),
    USER_AGENT   VARCHAR2(500),
    CREATED_AT   DATE DEFAULT SYSDATE
);

CREATE SEQUENCE LOGIN_LOG_SEQ START WITH 1 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER LOGIN_LOG_TRG
BEFORE INSERT ON LOGIN_LOG
FOR EACH ROW
BEGIN
    IF :NEW.LOG_ID IS NULL THEN
        SELECT LOGIN_LOG_SEQ.NEXTVAL INTO :NEW.LOG_ID FROM dual;
    END IF;
END;
/
